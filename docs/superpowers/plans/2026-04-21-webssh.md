# Admin-Only WebSSH Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an admin-only WebSSH page that is visible only to the first logged-in user, can open a backend-managed shell session, send commands, and stream command output back into the panel UI.

**Architecture:** Reuse the current Gin session login model and define administrator access as "current username matches the first user record". Add a small auth/capability endpoint plus a WebSocket-backed shell bridge on the backend, then add a lightweight frontend capability store, guarded route/menu entry, and a transcript-style terminal page that sends input and appends streamed output.

**Tech Stack:** Go 1.25, Gin, existing session helpers in `src/backend/internal/http/api`, Pinia, Vue 3, Vue Router, Vuetify 4, Vitest.

- [ ] **Step 5: Add the default SSH-backed bridge implementation shell**

```go
type sshShellBridge struct {
	settingService SettingService
	output         chan webSSHMessage
	stdin          io.WriteCloser
	client         *ssh.Client
	session        *ssh.Session
}

func newSSHBridge(settingService SettingService) webSSHBridge {
	return &sshShellBridge{
		settingService: settingService,
		output:         make(chan webSSHMessage, 16),
	}
}

func (b *sshShellBridge) Connect(ctx context.Context) error {
	host, err := b.settingService.GetWebSSHHost()
	if err != nil {
		return err
	}
	port, err := b.settingService.GetWebSSHPort()
	if err != nil {
		return err
	}
	username, err := b.settingService.GetWebSSHUsername()
	if err != nil {
		return err
	}
	password, err := b.settingService.GetWebSSHPassword()
	if err != nil {
		return err
	}

	config := &ssh.ClientConfig{
		User: username,
		Auth: []ssh.AuthMethod{ssh.Password(password)},
		HostKeyCallback: ssh.InsecureIgnoreHostKey(),
		Timeout: 10 * time.Second,
	}

	client, err := ssh.Dial("tcp", net.JoinHostPort(host, strconv.Itoa(port)), config)
	if err != nil {
		return err
	}
	b.client = client

	session, err := client.NewSession()
	if err != nil {
		return err
	}
	b.session = session

	stdout, err := session.StdoutPipe()
	if err != nil {
		return err
	}
	stderr, err := session.StderrPipe()
	if err != nil {
		return err
	}
	stdin, err := session.StdinPipe()
	if err != nil {
		return err
	}
	b.stdin = stdin

	if err := session.Shell(); err != nil {
		return err
	}

	go b.streamPipe(stdout, "stdout")
	go b.streamPipe(stderr, "stderr")
	return nil
}

func (b *sshShellBridge) Write(input string) error {
	_, err := io.WriteString(b.stdin, input)
	return err
}

func (b *sshShellBridge) Read() <-chan webSSHMessage {
	return b.output
}

func (b *sshShellBridge) Close() error {
	if b.session != nil {
		_ = b.session.Close()
	}
	if b.client != nil {
		_ = b.client.Close()
	}
	close(b.output)
	return nil
}
```

Keep the first pass minimal so the tests can exercise the handler wiring through a fake bridge before the real SSH connection logic is filled in.

```go
func (a *APIHandler) handleWebSSH(c *gin.Context) {
	username := GetLoginUser(c)
	isAdmin, err := a.UserService.IsFirstUser(username)
	if err != nil {
		c.JSON(http.StatusInternalServerError, Msg{Success: false, Msg: err.Error()})
		return
	}
	if !isAdmin {
		c.JSON(http.StatusForbidden, Msg{Success: false, Msg: "forbidden"})
		return
	}

	conn, err := websocket.Accept(c.Writer, c.Request, nil)
	if err != nil {
		return
	}
	defer conn.CloseNow()

	ctx := c.Request.Context()
	bridge := newSSHBridge(a.SettingService)
	defer bridge.Close()

	if err := bridge.Connect(ctx); err != nil {
		_ = wsjson.Write(ctx, conn, webSSHMessage{Type: "status", Status: "error", Error: err.Error()})
		return
	}

	_ = wsjson.Write(ctx, conn, webSSHMessage{Type: "status", Status: "connected"})

	go func() {
		for msg := range bridge.Read() {
			_ = wsjson.Write(ctx, conn, msg)
		}
	}()

	for {
		var incoming webSSHMessage
		if err := wsjson.Read(ctx, conn, &incoming); err != nil {
			return
		}
		if incoming.Type == "input" {
			if err := bridge.Write(incoming.Data); err != nil {
				_ = wsjson.Write(ctx, conn, webSSHMessage{Type: "status", Status: "error", Error: err.Error()})
				return
			}
		}
	}
}
```

```ts
export type WebSSHConnectionStatus = 'disconnected' | 'connecting' | 'connected' | 'error'

export interface WebSSHIncomingMessage {
  type: 'status' | 'output' | 'close'
  status?: string
  stream?: 'stdout' | 'stderr'
  data?: string
  error?: string
}

export interface WebSSHSessionState {
  connectionStatus: WebSSHConnectionStatus
  transcript: string
  lastError: string
}

export const createSessionState = (): WebSSHSessionState => ({
  connectionStatus: 'disconnected',
  transcript: '',
  lastError: '',
})

export const appendMessage = (state: WebSSHSessionState, chunk: string): void => {
  state.transcript += chunk
}

export const reduceIncomingMessage = (state: WebSSHSessionState, message: WebSSHIncomingMessage): void => {
  if (message.type === 'output') {
    appendMessage(state, message.data ?? '')
    return
  }

  if (message.type === 'status') {
    state.connectionStatus = (message.status as WebSSHConnectionStatus) ?? 'error'
    state.lastError = message.error ?? ''
  }
}
```

```ts
let socket: WebSocket | null = null

const connect = () => {
  state.value.connectionStatus = 'connecting'
  socket = new WebSocket(`${window.location.origin.replace('http', 'ws')}${(window as any).BASE_URL}api/webssh/ws`)
  socket.onmessage = event => {
    reduceIncomingMessage(state.value, JSON.parse(event.data))
  }
  socket.onerror = () => {
    state.value.connectionStatus = 'error'
    state.value.lastError = 'WebSSH connection failed'
  }
  socket.onclose = () => {
    if (state.value.connectionStatus !== 'error') {
      state.value.connectionStatus = 'disconnected'
    }
  }
}

const sendCommand = () => {
  if (!socket || socket.readyState !== WebSocket.OPEN || !command.value) return
  socket.send(JSON.stringify({ type: 'input', data: `${command.value}\n` }))
  command.value = ''
}
```
